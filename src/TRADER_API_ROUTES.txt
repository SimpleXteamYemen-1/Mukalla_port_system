// Get trader dashboard statistics
app.get("/make-server-85dcafc8/trader-stats", async (c) => {
  try {
    const traderEmail = c.req.query('email');
    
    if (!traderEmail) {
      return c.json({ success: false, error: "Trader email required" }, 400);
    }
    
    const containers = await kv.getByPrefix("container:");
    const dischargeRequests = await kv.getByPrefix("discharge:");
    const notifications = await kv.getByPrefix("notification:");
    
    // Filter trader's data
    const traderContainers = containers.filter((c: any) => c.traderEmail === traderEmail);
    const traderDischarges = dischargeRequests.filter((d: any) => d.traderEmail === traderEmail);
    const traderNotifications = notifications.filter((n: any) => n.traderEmail === traderEmail && !n.read);
    
    const arrived = traderContainers.filter((c: any) => c.status === 'arrived').length;
    const stored = traderContainers.filter((c: any) => c.status === 'assigned').length;
    const ready = traderContainers.filter((c: any) => c.status === 'ready_discharge').length;
    
    const statusChanges = traderNotifications.filter((n: any) => n.type === 'status_change').length;
    
    return c.json({
      success: true,
      stats: {
        arrived,
        stored,
        readyForDischarge: ready,
        unreadNotifications: traderNotifications.length,
        pendingDischarges: traderDischarges.filter((d: any) => d.status === 'pending').length,
        statusChangeAlerts: statusChanges
      }
    });
  } catch (error) {
    console.error("Error fetching trader stats:", error);
    return c.json({ success: false, error: String(error) }, 500);
  }
});

// Submit discharge request
app.post("/make-server-85dcafc8/discharge-request", async (c) => {
  try {
    const { containerId, traderEmail, traderName, requestedDate, notes } = await c.req.json();
    
    const container = await kv.get(containerId);
    if (!container) {
      return c.json({ success: false, error: "Container not found" }, 404);
    }
    
    if (container.traderEmail !== traderEmail) {
      return c.json({ success: false, error: "Unauthorized" }, 403);
    }
    
    const allRequests = await kv.getByPrefix("discharge:");
    const requestNumber = String(allRequests.length + 1).padStart(4, '0');
    const requestId = `discharge:DR-${requestNumber}`;
    
    const dischargeRequest = {
      id: requestId,
      requestId: `DR-${requestNumber}`,
      containerId: container.containerId || container.id,
      traderEmail,
      traderName,
      vesselName: container.vesselName,
      storageLocation: container.assignedStorage,
      requestedDate,
      notes,
      status: 'pending',
      submittedAt: new Date().toISOString()
    };
    
    await kv.set(requestId, dischargeRequest);
    
    const notifId = `notification:${Date.now()}-${traderEmail}`;
    await kv.set(notifId, {
      id: notifId,
      traderEmail,
      type: 'discharge_submitted',
      title: 'Discharge Request Submitted',
      message: `Your discharge request ${dischargeRequest.requestId} has been submitted`,
      timestamp: new Date().toISOString(),
      read: false
    });
    
    return c.json({ success: true, request: dischargeRequest });
  } catch (error) {
    console.error("Error submitting discharge request:", error);
    return c.json({ success: false, error: String(error) }, 500);
  }
});

// Get trader notifications
app.get("/make-server-85dcafc8/trader-notifications", async (c) => {
  try {
    const traderEmail = c.req.query('email');
    
    if (!traderEmail) {
      return c.json({ success: false, error: "Trader email required" }, 400);
    }
    
    const notifications = await kv.getByPrefix("notification:");
    const traderNotifications = notifications.filter((n: any) => n.traderEmail === traderEmail);
    
    traderNotifications.sort((a: any, b: any) => 
      new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime()
    );
    
    return c.json({ success: true, notifications: traderNotifications });
  } catch (error) {
    console.error("Error fetching trader notifications:", error);
    return c.json({ success: false, error: String(error) }, 500);
  }
});

// Mark notification as read
app.post("/make-server-85dcafc8/mark-notification-read", async (c) => {
  try {
    const { notificationId } = await c.req.json();
    
    const notification = await kv.get(notificationId);
    if (!notification) {
      return c.json({ success: false, error: "Notification not found" }, 404);
    }
    
    notification.read = true;
    await kv.set(notificationId, notification);
    
    return c.json({ success: true, notification });
  } catch (error) {
    console.error("Error marking notification as read:", error);
    return c.json({ success: false, error: String(error) }, 500);
  }
});
